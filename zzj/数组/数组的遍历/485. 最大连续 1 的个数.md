```python
class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        cnt = 0
        max = 0
        for num in nums:
            if num == 1:
                cnt = cnt + 1
            else:
                if cnt > max:
                    max = cnt
                cnt = 0
        if cnt > max:
            max = cnt
            cnt = 0
        return max
```
